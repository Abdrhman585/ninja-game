import time import random class LifeSim: def __init__(self): self.stats = { "money": 500, "hunger": 0, # من 0 إلى 100 "energy": 100, # من 0 إلى 100 "skills": 0, # مستوى الخبرة "day": 1 } self.inventory = [] self.job = "عامل بسيط" self.is_running = True def display_ui(self): print("\n" + "="*30) print(f" اليوم: {self.stats['day']} | الوظيفة: {self.job}") print(f" المال: {self.stats['money']}$ | الخبرة: {self.stats['skills']}") print(f" الجوع: {self.stats['hunger']}% | الطاقة: {self.stats['energy']}%") print("="*30) def work(self): if self.stats['energy'] < 20: print("! أنت متعب جداً للعمل، اذهب للنوم.") return # الراتب يعتمد على مستوى المهارة salary = 50 + (self.stats['skills'] * 10) self.stats['money'] += salary self.stats['energy'] -= 30 self.stats['hunger'] += 20 self.stats['skills'] += 1 print(f" تم العمل بنجاح! حصلت على {salary}$ وزادت خبرتك.") def store(self): print("\n--- المتجر ---") print("1. وجبة سريعة (20$) - تقلل الجوع 30%") print("2. وجبة صحية (50$) - تقلل الجوع 60% وتزيد الطاقة 10%") print("3. دورة تدريبية (200$) - تزيد الخبرة 5 مستويات") choice = input("ماذا تريد أن تشتري؟ (أو 0 للعودة): ") if choice == "1" and self.stats['money'] >= 20: self.stats['money'] -= 20 self.stats['hunger'] = max(0, self.stats['hunger'] - 30) print(" يم! الأكل كان لذيذأ.") elif choice == "3" and self.stats['money'] >= 200: self.stats['money'] -= 200 self.stats['skills'] += 5 print(" زادت مهاراتك بشكل ملحوظ!") else: print(" فشلت العملية (ربما لا تملك المال الكافي).") def sleep(self): print(" نائم الآن... Zzz") time.sleep(1) self.stats['energy'] = 100 self.stats['hunger'] += 30 self.stats['day'] += 1 print(" صباح جديد! استعدت طاقتك بالكامل.") def check_death(self): if self.stats['hunger'] >= 100: print("\n ماتت شخصيتك بسبب الجوع الشديد. جيم أوفر!") self.is_running = False def run(self): print("مرحباً بك في محاكي الحياة الواقعية!") while self.is_running: self.display_ui() print("1. الذهاب للعمل") print("2. زيارة المتجر") print("3. النوم (بداية يوم جديد)") print("4. خروج") action = input("ما هي خطتك القادمة؟ ") if action == "1": self.work() elif action == "2": self.store() elif action == "3": self.sleep() elif action == "4": self.is_running = False self.check_death() # تشغيل اللعبة game = LifeSim() game.run()
